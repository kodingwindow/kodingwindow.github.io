{{ site.asmfile }}
{% highlight nasm %}
section .data
h times 60 db "Hello, World!",10

section .text
global _start
_start:
    mov eax,4
    mov ebx,1
    mov ecx,h
    mov edx,60
    int 80h

;Exit System Call
    mov eax,1
    mov ebx,0
    int 80h
{% endhighlight %}
