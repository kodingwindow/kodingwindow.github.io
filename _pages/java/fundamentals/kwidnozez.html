class KW
{
    public static void main(String args[])
    {
        boolean a=true,b=false;
        System.out.println("———————————————————————————————————————————");
        System.out.println("Program to perform the boolean operations");
        System.out.println("———————————————————————————————————————————");
        System.out.println(a+" |  "+b+"\t| "+(a|b));
        System.out.println(a+" &  "+b+"\t| "+(a&b));
        System.out.println(a+" ^  "+b+"\t| "+(a^b));
        System.out.println(a+" || "+b+"\t| "+(a||b));
        System.out.println(a+" && "+b+"\t| "+(a&&b));
        System.out.println(a+" == "+b+"\t| "+(a==b));
        System.out.println(a+" != "+b+"\t| "+(a!=b));
        System.out.println("!("+a+" || "+b+")| "+!(a||b));
        System.out.println("———————————————————————————————————————————");
    }
}
