{{ site.cfile }}
{% highlight c %}
#include 
int main()
{
    int a = 0, b = 1;
    printf("———————————————————————————————————————————");
    printf("\nProgram to perform relational operations");
    printf("\n———————————————————————————————————————————");
    printf("\nValue of a     | %d ", a);
    printf("\nValue of b     | %d ", b);
    printf("\nValue of %d!=%d  | %d ", a, b, (a != b));
    printf("\nValue of %d> %d  | %d ", a, b, (a > b));
    printf("\nValue of %d>=%d  | %d ", a, b, (a >= b));
    printf("\nValue of %d==%d  | %d ", a, b, (a == b));
    printf("\nValue of %d< %d  | %d ", a, b, (a < b));
    printf("\n———————————————————————————————————————————\n");
    return 0;
}
