print("________________________________________")
print("Program to print list of all leap years")
print("________________________________________")
y=int(input("How many years you want to print "))
if(y<1):
    print("Enter valid number of years")
else:
    lc=0
    for i in range(y):
        year=2018+i
        if(year%4==0):
            if(year%100==0):
                if(year%400==0):
                    print("{0} is leap year".format(year))
                    lc=lc+1
                else:
                    print("{0} is not leap year".format(year))
            else:
                print("{0} is leap year".format(year))
                lc=lc+1
        else:
            print("{0} is not leap year".format(year))
    print("Total leap years between 2018 to {0} is".format(year),lc)
print("________________________________________")
