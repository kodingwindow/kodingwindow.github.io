(write-line "Program to check a number is +ve or -ve")
(write-line "———————————————————————————————————————————")
(princ "Enter the number ")
(setq a(read))
(if(= a 0)
(format t "~%~d is neither +ve nor -ve "a))
(if(> a 0)
(format t "~%~d is +ve number "a))
(if(< a 0)
(format t "~%~d is -ve number "a))
(terpri)(format t "———————————————————————————————————————————")
