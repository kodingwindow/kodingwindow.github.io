echo "———————————————————————————————————————————"
echo "Script to check a given number is +ve or -ve "
echo "———————————————————————————————————————————"
echo "Enter the number \c"; read a
b=0
if [ $a -gt $b ];then
    echo "Entered number is positive"
elif [ $a -eq $b ];then
    echo "Entered number is zero"
else
    echo "Entered number is negative"
fi
echo "———————————————————————————————————————————"
{% endhighlight %}
