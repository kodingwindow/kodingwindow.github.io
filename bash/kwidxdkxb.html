Title: Shell script to check the equality of given numbers
{{ site.shfile }}
{% highlight shell %}
echo "———————————————————————————————————————————"
echo "Script to check the equality of given numbers"
echo "———————————————————————————————————————————"
echo "Enter the 1st number \c"; read a
echo "Enter the 2nd number \c"; read b
if [ $a -gt $b ];then
    echo "$a is greater than $b"
elif [ $a -eq $b ];then
    echo "$a is equal to $b"
else
    echo "$a is less than $b"
fi
echo "———————————————————————————————————————————"
{% endhighlight %}

Title: Shell script to check the equality of given strings
{{ site.shfile }}
{% highlight shell %}
echo "———————————————————————————————————————————"
echo "Script to check the equality of given strings"
echo "———————————————————————————————————————————"
echo "Enter the 1st string \c"; read a
echo "Enter the 2nd string \c"; read b 
if [ $a = $b ];then
    echo "$a is equal to $b"
else
    echo "$a is not equal to $b"
fi
echo "———————————————————————————————————————————"
{% endhighlight %}
