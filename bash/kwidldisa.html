Title: Shell script to perform the arithmetic operations
{{ site.shfile }}
{% highlight shell %}
echo "10+20  = \c"; expr 10 + 20
echo "10-20  = \c"; expr 10 - 20
echo "10\*20 = \c"; expr 10 \* 20
echo "10/20  = \c"; expr 10 / 20
{% endhighlight %}
{{ site.output }}
{{ site.runsh }}
10+20  = 30
10-20  = -10
10\*20 = 200
10/20  = 0

Title: Shell script to perform the arithmetic operations
{{ site.shfile }}
{% highlight shell %}
echo $((10 + 20))
echo $((10 - 20))
echo $((10 \* 20))
echo $((10 / 20))
{% endhighlight %}
{{ site.output }}
{{ site.runsh }}
30
-10
200
0
