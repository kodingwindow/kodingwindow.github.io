import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;
public class KW
{
    public static void main(String[] args) 
    {
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        WebDriver driver = new ChromeDriver(); 
        driver.navigate().to("https://www.wikipedia.org/");
        //driver.findElement(By.xpath("//*[@id='searchLanguage']/option[20]")).click();
        Select dropdown = new Select(driver.findElement(By.id("searchLanguage")));
        //Select dropdown = new Select(driver.findElement(By.xpath("//*[@id='searchLanguage']")));
        dropdown.selectByIndex(19);
        //driver.close();
        System.exit(0);
    }
}
