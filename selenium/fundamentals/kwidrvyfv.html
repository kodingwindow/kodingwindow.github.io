Title: Selenium Java to launch the Chrome browser and navigate to a given URL

{% highlight java %}
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class KW 
{
    public static void main(String[] args) 
    {
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        WebDriver driver = new ChromeDriver();     
        driver.navigate().to("https://kodingwindow.com/testapp/");
        driver.close();
        System.exit(0);
    }   	
}

Title: Selenium Java to launch the Chromium browser and navigate to a given URL
{{ site.javafile }}
{% highlight java %}
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;

public class KW 
{
    public static void main(String[] args) 
    {
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        ChromeOptions co = new ChromeOptions();
        co.setBinary("D:\\Selenium\\chrome-win\\chrome.exe");
        WebDriver driver = new ChromeDriver(co);
        driver.manage().window().maximize();
        driver.navigate().to("https://kodingwindow.com/testapp/");
        driver.close();
    }       
}
{% endhighlight %}
