Title: Selenium Java to get domain name, URL, and page title using JavascriptExecutor
{{ site.javafile }}
{% highlight java %}
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
public class KW
{
    public static void main(String[] args) 
    {   
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.get("http://newtours.demoaut.com");
        
        JavascriptExecutor jse = (JavascriptExecutor)driver;
        String domain = jse.executeScript("return document.domain;").toString();
        System.out.println("Domain Name - "+domain);
        
        String url = jse.executeScript("return document.URL;").toString();
        System.out.println("URL - "+url);
        
        String title = jse.executeScript("return document.title;").toString();
        System.out.println("Page Title - "+title);
        
        //driver.close();
    }
}
{% endhighlight %}
{{ site.output }}
Domain Name - newtours.demoaut.com
URL - http://newtours.demoaut.com/
Page Title - Welcome: Mercury Tours
Title: Selenium Java to generate alert and navigate to another page using JavascriptExecutor
{{ site.javafile }}
{% highlight java %}
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
public class KW
{
    public static void main(String[] args) 
    {   
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.get("http://newtours.demoaut.com");
        
        JavascriptExecutor jse = (JavascriptExecutor)driver;
        jse.executeScript("window.location = 'http://newtours.demoaut.com/mercurysignon.php'");
        
        jse.executeScript("alert('Navigated from home page');");
        
        //driver.close();
    }
}

Title: Selenium Java to scroll down the page at the bottom using JavascriptExecutor
{{ site.javafile }}
{% highlight java %}
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
public class KW
{
    public static void main(String[] args) 
    {   
        System.setProperty("webdriver.chrome.driver", "D://Selenium//Drivers//chromedriver.exe");
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.get("http://newtours.demoaut.com");
        
        JavascriptExecutor jse = (JavascriptExecutor)driver;
        jse.executeScript("window.scrollTo(0, document.body.scrollHeight)");
        
        //jse.executeScript("window.scrollBy(0, 500)"); //scrollBy(x-pixels,y-pixels)
        //driver.close();
    }
}
