Title: {{ page.title }}
{{ site.javafile }}
{% highlight java %}
public class KW
{
    public static void main()
    {
        System.out.println("Welcome to Kodingwindow");
    }
    public static void main(String args[])
    {
        System.out.println("Hello, World!");
        main();
    }
}
{% endhighlight %}
{{ site.output }}
{{ site.runjava }}
Hello, World!
Welcome to Kodingwindow
kodingwindow@kw:~$

{% highlight java %}
public class KW
{
    public static void main(String args) 
    {
        System.out.println("Welcome to "+args);
    }
    public static void main(String[] args) 
    {
        System.out.println("Hello, World!");
        main("Kodingwindow");
    }
}
{% endhighlight %}
