arr = Array(1000,900,800,700,600,500,400,300,200,100)
For i = LBound(arr) to UBound(arr)
    For j = LBound(arr) to UBound(arr) - 1
        If arr(j) > arr(j + 1) Then
            temp = arr(j + 1)
            arr(j + 1) = arr(j)
            arr(j) = temp
        End If
    Next
Next

For i = LBound(arr) To UBound(arr)
    sorted = sorted & arr(i)&" "
Next
print "Sorted Array Is "&sorted
{% endhighlight %}
