using System;
using System.Threading;
namespace Kodingwindow
{
    class KWThread
    {
        public void thread()
        {
            for(int i = 100; i < 105; i++)
            {
                Console.Write(i+" ");
            }
        }
    }
    public class Program
    {
        public static void thread()
        {
            for(int i = 0; i < 5; i++)
            {
                Console.Write(i+" ");
            }
        }
        public static void Main(string[] args)
        {
            Console.WriteLine("———————————————————————————————————————————");
            Thread t = Thread.CurrentThread;
            t.Name = "Main Thread";
            Console.WriteLine(t.Name);

            Thread t1 = new Thread(new ThreadStart(Program.thread));
            Thread t2 = new Thread(new ThreadStart(Program.thread));
            t1.Start();
            t2.Start();

            KWThread kwt = new KWThread();
            Thread t3 = new Thread(new ThreadStart(kwt.thread));
            Thread t4 = new Thread(new ThreadStart(kwt.thread));
            t3.Start();
            t4.Start();
            Console.WriteLine("\n———————————————————————————————————————————");
        }
    }
}
