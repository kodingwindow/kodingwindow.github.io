Title: C# program to convert uppercase to a lowercase string using ASCII values
{{ site.csfile }}
{% highlight csharp %}
using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            string s1= "KODIGWINDOW";
            string s2 = "";
            for(int i=0; i < s1.Length; i++)
            {
                if (s1[i] >= 65 && s1[i] <= 92)
                    s2 += (char)(s1[i] + 32);
                else  
                    s2 += s1[i];  
            }
            Console.WriteLine(s2);
        }
    }
}
