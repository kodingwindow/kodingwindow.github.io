using System;
namespace Kodingwindow
{
    public class Computer
    {
        public String mouseType;
        public String mouseCmp;
        public int mouseId;

        public Computer()
        {
            Console.WriteLine("Constructor");
        }
        public Computer(String t, String c, int i)
        {
            mouseType = t;
            mouseCmp = c;
            mouseId = i;
        }
        ~Computer()
        {
            Console.WriteLine("Destructor");
        }
        public void display()
        {
            Console.WriteLine("Mouse Type: " + mouseType + "\nMouse Cmp:  " + mouseCmp + "\nMouse Id:   " + mouseId);
        }
    }
    public class Program
    {
        public static void Main(string[] args)
        {
            Console.WriteLine("———————————————————————————————————————————");
            Console.WriteLine("Program to use a constructor and destructor");
            Console.WriteLine("———————————————————————————————————————————");
            Computer c1 = new Computer();
            Computer c2 = new Computer("Wireless", "Lenovo", 145221);
            c2.display();
            c1 = null;
            GC.Collect();
            Console.WriteLine("———————————————————————————————————————————");
        }
    }
}
