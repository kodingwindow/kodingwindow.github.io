using System;
namespace Kodingwindow
{
    public class Fruit
    {
        private string fname;
        private string fcolor;
        public string FruitName
        {
            get
            {
                return fname;
            }
            set
            {
                fname = value;
            }
        }

        public string FruitColor
        {
            get
            {
                return fcolor;
            }
            set
            {
                fcolor = value;
            }
        }

        public override string ToString()
        {
            return "Fruit Name - " + FruitName + "\nFruit Color - " + FruitColor;
        }
    }
    public class Program
    {
        public static void Main(string[] args)
        {
            Fruit f = new Fruit();
            f.FruitName = "Apple";
            f.FruitColor = "Red";
            Console.WriteLine("———————————————————————————————————————————");
            Console.WriteLine("Program to use the accessors properties");
            Console.WriteLine("———————————————————————————————————————————");
            Console.WriteLine("Fruit Name - " + f.FruitName);
            Console.WriteLine("Fruit Color - " + f.FruitColor);
            Console.WriteLine(f);
            Console.WriteLine("———————————————————————————————————————————");
        }
    }
}
