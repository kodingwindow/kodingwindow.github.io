Title: C# program to demonstrate the use of break statement
{{ site.csfile }}
{% highlight csharp %}
using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            for(int i=0;i<10;i++)
            {
                if(i==5)
                {
                    break;
                }
                Console.WriteLine(i);    
            }
        }
    }
}
