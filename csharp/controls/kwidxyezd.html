using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            int a;
            Console.WriteLine("———————————————————————————————————————————");
            Console.WriteLine("Program to check a given number is +ve/-ve");
            Console.WriteLine("———————————————————————————————————————————");
            Console.Write("Enter the number ");
            a = Convert.ToInt32(Console.ReadLine());
            if(a < 0)
            {
                Console.WriteLine("\nEntered number is negative");
            }
            else if(a > 0)
            {
                Console.WriteLine("\nEntered number is positive");
            }
            else
            {
                Console.WriteLine("\nEntered number is zero");
            }
            Console.WriteLine("———————————————————————————————————————————");
        }
    }
}
