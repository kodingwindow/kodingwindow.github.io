using System;
using System.Collections.Generic;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            HashSet hs=new HashSet();
            hs.Add("Cherry");
            hs.Add("Apple");
            hs.Add("Orange");
            hs.Add("Mango");
            hs.Add("Grapes");
            hs.Add("Cherry");
            hs.Add("Apple");
            hs.Add("Pineapple");
            hs.Add("Cherry");
            
            foreach(string s in hs)
            {
                Console.Write(s+" ");
            }
            Console.WriteLine("\nSize: "+hs.Count);
        }
    }
}

Title: {{ page.title }}: Remove() and Clear() methods
{{ site.csfile }}
{% highlight csharp %}
using System;
using System.Collections.Generic;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            HashSet hs=new HashSet();
            hs.Add("Cherry");
            hs.Add("Apple");
            hs.Add("Orange");
            hs.Add("Mango");
            hs.Add("Grapes");
            hs.Add("Cherry");
            hs.Add("Apple");
            hs.Add("Pineapple");
            hs.Add("Cherry");
            
            hs.Remove("Apple");
            foreach(string s in hs)
            {
                Console.Write(s+" ");
            }
            hs.Clear();
            Console.WriteLine("\nSize: "+hs.Count);
        }
    }
}
{% endhighlight %}
{{ site.output }}
Cherry Orange Mango Grapes Pineapple 
Size: 0
