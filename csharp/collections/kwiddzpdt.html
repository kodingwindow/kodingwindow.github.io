{% highlight csharp %}
using System;
using System.Collections.Generic;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            SortedSet ss=new SortedSet();
            ss.Add("Cherry");
            ss.Add("Apple");
            ss.Add("Orange");
            ss.Add("Mango");
            ss.Add("Grapes");
            ss.Add("Cherry");
            ss.Add("Apple");
            ss.Add("Pineapple");
            ss.Add("Cherry");
            
            ss.Remove("Apple");
            foreach(string s in ss)
            {
                Console.Write(s+" ");
            }
            ss.Clear();
            Console.WriteLine("\nSize: "+ss.Count);
        }
    }
}
{% endhighlight %}
{{ site.output }}
Cherry Grapes Mango Orange Pineapple 
Size: 0
