using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            int i,j;
            for(i=1;i<=7;i++)
            {
                for(j=1;j<=7-i;j++)
                {
                    Console.Write(" ");
                }
                for(j=1;j<=i;j++)
                {
                    Console.Write("* "); 
                }
                Console.Write("\n");
            }
            for(i=6;i>=1;i--)
            {
                for(j=1;j<=6-i;j++)
                {
                    Console.Write(" ");
                }
                for(j=1;j<=i;j++)
                {
                    Console.Write(" *" ); 
                }
                Console.Write("\n");
            }
        }
    }
}
