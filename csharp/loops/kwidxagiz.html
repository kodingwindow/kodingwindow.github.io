using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            Console.WriteLine("———————————————————————————————————————————");
            Console.WriteLine("Program to use the control statements");
            Console.WriteLine("———————————————————————————————————————————");
            int i = 0;
            do
            {
                i++;
                if (i == 9)
                {
                    break;
                }
                if(i == 6)
                {
                    continue;
                }
                Console.WriteLine(i); 
            } while(i <= 10);
            goto Execute;
            Console.WriteLine("———————————————————————————————————————————");
            Execute:
            Console.WriteLine("Executed goto statement");
        }
    }
}
