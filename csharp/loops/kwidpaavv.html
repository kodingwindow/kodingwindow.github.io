Title: C# program to print the right half star pyramid pattern
{{ site.csfile }}
{% highlight csharp %}
using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            int i,j;
            for(i=1;i<=10;i++)
            {
                for(j=1;j<=i;j++)
                {
                    Console.Write("*");
                }
                Console.Write("\n");
            }
        }
    }
}
{% endhighlight %}
