using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            string s=null;
            try
            {
                Console.WriteLine("Length: "+s.Length);
            }
            catch(Exception e)
            {
                Console.WriteLine("Exception caught");
            }
            finally
            {
                Console.WriteLine("Finally block will always execute");
            }
        }
    }
}
{% endhighlight %}
