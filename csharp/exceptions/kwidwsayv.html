using System;
namespace Kodingwindow
{
    public class Program
    {
        public static void Main(string[] args)
        {
            int n=515, d=0, result=0;
            try
            {
                Console.WriteLine(result=n/d);
            }
            catch(DivideByZeroException e)
            {
                Console.WriteLine(e.Message);
            }
        }
    }
}
