{% highlight cpp %}
#include 
using namespace std;
class kw
{
public:
    kw()
    {
        cout<<"\nConstructor is called";
    }
    ~kw()
    {
        cout<<"\nDestructor is called";
    }
};
int main()
{
    kw k;
    kw();
    k.~kw();
}
{% endhighlight %}
