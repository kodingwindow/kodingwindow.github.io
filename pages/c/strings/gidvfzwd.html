int main()
{
    char s[20];
    int i;
    printf("———————————————————————————————————————————");
    printf("\nProgram to convert lower to an uppercase string");
    printf("\n———————————————————————————————————————————");
    printf("\nEnter the string ");
    scanf("%s", s);
    for (i = 0; i <= strlen(s); i++)
    {
        if (s[i] >= 97 && s[i] <= 122)
            s[i] = s[i] - 32;
    }
    printf("\nThe uppercase string is %s", s);
    printf("\n———————————————————————————————————————————\n");
    return 0;
}

{{ page.title }} using toupper() function
{{ site.cfile }}
{%- highlight c -%}
#include 
int main()
{
    char s[20];
    int i;
    printf("———————————————————————————————————————————");
    printf("\nProgram to convert lower to an uppercase string");
    printf("\n———————————————————————————————————————————");
    printf("\nEnter the string ");
    scanf("%s", s);
    for (i = 0; s[i]; i++)
        s[i] = toupper(s[i]);
    printf("\nThe uppercase string is %s", s);
    printf("\n———————————————————————————————————————————\n");
    return 0;
}
