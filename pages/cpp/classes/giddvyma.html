#include 
using namespace std;
class gd
{
public:
    gd()
    {
        cout<<"\nConstructor is called";
    }
    ~gd()
    {
        cout<<"\nDestructor is called";
    }
};
int main()
{
    gd k;
    gd();
    k.~gd();
}
