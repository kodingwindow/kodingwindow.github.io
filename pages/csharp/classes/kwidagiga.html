public class Fruit
{
    private string fname = "";
    private string fcolor = "";
    public string FruitName
    {
        get
        {
            return fname;
        }
        set
        {
            fname = value;
        }
    }

    public string FruitColor
    {
        get
        {
            return fcolor;
        }
        set
        {
            fcolor = value;
        }
    }

    public override string ToString()
    {
        return "Fruit Name - " + FruitName + "\nFruit Color - " + FruitColor;
    }
}
