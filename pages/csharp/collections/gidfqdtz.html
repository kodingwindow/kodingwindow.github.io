using System.Collections.Generic;

HashSet hs = new HashSet();
hs.Add("Cherry");
hs.Add("Apple");
hs.Add("Orange");
hs.Add("Mango");
hs.Add("Grapes");
hs.Add("Cherry");
hs.Add("Apple");

foreach (string s in hs)
{
    Console.Write(s + " ");
