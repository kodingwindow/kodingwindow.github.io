{{ page.title }}
{{ site.vbsfile }}
{%- highlight vb -%}
Set objFSO = CreateObject("Scripting.FileSystemObject")
Set objFolder = objFSO.GetFolder("D:\VBScript")
counter = 0
For Each File In objFolder.Files
    If objFSO.GetExtensionName(File) = "txt" Then
        counter = counter + 1
    End If
Next
print "Total txt File " & counter
{%- endhighlight -%}
{{ site.output }}

Total Files 15
