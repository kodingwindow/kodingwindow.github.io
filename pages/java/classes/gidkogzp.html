{{ site.javafile }}
{%- highlight java -%}
class GD
{
    int a;
    GD()
    {
        this(30);
    }
    GD(int a)
    {
        this.a=a;
        System.out.println(a);
    }
    public static void main(String args[])
    {
        GD k1=new GD();
    }
}
{%- endhighlight -%}
